我们设置baseURL和请求超时时间。

import axios from 'axios'
import store from '../store'
import i18n from '@/utils/i18n'
import { ElNotification, ElMessage } from 'element-plus'

// 创建axios实例
const request = axios.create({
  baseURL: import.meta.env.PROD ? import.meta.env.BASE_URL : '/', // api 的 base_url
  timeout: 12000000 // 请求超时时间
})

// request拦截器
request.interceptors.request.use(
  config => {
      let temp: {
        'OCS-APIRequest': boolean;
        'xyw-Client': boolean;
        'Content-Type': string;
        Authorization?: string;
      } = {
        'OCS-APIRequest': true,
        'xyw-Client': true,
        'Content-Type': 'application/json'
      }
      const token = localStorage.getItem('xywToken')
      if (token) {
        temp.Authorization = token
      }
      config.headers = Object.assign(
        temp,
        config.headers
      )
    return config
  },
  error => {
    // Do something with request error
    Promise.reject(error)
  }
)

// response 拦截器
request.interceptors.response.use(
  (response: any) => {
    const code = response.status
    if (code < 200 || code > 300) {
      const errmsg = response.message
      ElNotification({
        type: 'error',
        title: errmsg
      })
      return Promise.reject('error')
    } else {
      return response
    }
  },
  error => {
    if (error &&
      error.config &&
      error.config.headers &&
      error.config.headers.not_show_err) {
      return Promise.reject(error)
    }

    let code = 0
    try {
      code = error.response.status
    } catch (e) {
      if (error.toString().indexOf('Error: timeout') !== -1 ||
      (error.toString().indexOf('Error: Network Error') !== -1)) {
        ElMessage.error(i18n.global.t('errorMessage.serverOffline'))
        return Promise.reject(error)
      }
    }
    if (code === 401) {
      store.dispatch('LogOut')
    } else {
      let errorMsg = error.toString()
      if (error &&
        error.response &&
        error.response.data) {
        errorMsg = error.response.data
        ElNotification({
          type: 'error',
          title: errorMsg,
          duration: 3000
        })
    }
    return Promise.reject(error)
  }
})

export default request
